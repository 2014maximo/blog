-- Calcular el número de días trancurridos entre la fecha actual y su fecha de cumpleaños -- Calcular el número de meses trancurridos entre la fecha actual y su fecha de nacimiento -- Calcular el número de años trancurridos entre la fecha actual y su fecha de nacimiento -- Calcular el número de días transcurridos entre la fecha actual y la fecha de nacimiento

SELECT DATEDIFF(DAY, '1983-02-10', CURRENT_TIMESTAMP) AS DIAS_TRANSCURRIDOS
GO
SELECT DATEDIFF(MONTH, '1983-02-10', CURRENT_TIMESTAMP) AS MESES_TRANSCURRIDOS
GO
SELECT DATEDIFF(YEAR, '1983-02-10', CURRENT_TIMESTAMP) AS AÑOS_TRANSCURRIDOS
GO
